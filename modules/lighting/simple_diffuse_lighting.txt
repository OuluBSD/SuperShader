// Simple Diffuse Lighting Module
// Calculates basic diffuse lighting based on normal and light direction

// Calculate diffuse lighting using Lambert's cosine law
vec3 calculate_diffuse_lighting(vec3 normal, vec3 light_direction, vec3 light_color) {
    float diff = max(dot(normal, light_direction), 0.0);
    vec3 diffuse = diff * light_color;
    return diffuse;
}

// Calculate diffuse lighting with material properties
vec3 calculate_diffuse_with_material(vec3 normal, vec3 light_direction, vec3 light_color, vec3 albedo) {
    float diff = max(dot(normal, light_direction), 0.0);
    vec3 diffuse = diff * light_color * albedo;
    return diffuse;
}
